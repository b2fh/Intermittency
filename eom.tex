\begin{equation}

\begin{equation}
  \sqrt{|g|} = a^3
\end{equation}

\begin{equation}

\subsection{Conformal Time}

\begin{equation}
  ds^2 = a^2\left(-d\tau^2 + dx^2\right)
\end{equation}

\begin{equation}
  R = \frac{6}{a^2}\left(\frac{a''}{a}\right)
\end{equation}
and
\begin{equation}
  \sqrt{|g|} = a^4
